# Integration of Transylvania
country_event = {
	id = hungary_flavor_territory.1
	title = hungary_flavor_territory.1.t
	desc = hungary_flavor_territory.1.d
	picture = GFX_report_event_HUN_transylvania
	
	trigger = {
		has_country_flag = HUN_can_integrate
		NOT = { has_country_flag = HUN_transylvania_integration_off }
		NOT = { has_country_flag = HUN_transylvania_integration_on  }
		tag = HUN
		OR = {
			AND = {
				owns_state = 82		#Banat
				NOT = { 82 = { is_core_of = ROOT } }	#Banat
			}
			AND = {
				owns_state = 83		#Southern Partium
				NOT = { 83 = { is_core_of = ROOT } }	#Southern Partium
			}
			AND = {
				owns_state = 84		#Transylvania
				NOT = { 84 = { is_core_of = ROOT } }	#Transylvania
			}
			AND = {
				owns_state = 752		#Maramaros
				NOT = { 752 = { is_core_of = ROOT } }	#Maramaros
			}
			AND = {
				owns_state = 753		#Northern Partium
				NOT = { 753 = { is_core_of = ROOT } }	#Northern Partium
			}
		}
	}

	mean_time_to_happen = {
		days = 7
	}
	
	#Integrate
	option = {
		name = hungary_flavor_territory.1.a
		custom_effect_tooltip = HUN_transylvania_will_be_gradually_integrated
		hidden_effect = { 
			set_country_flag = HUN_transylvania_integration_on 
			if = {
				limit = { owns_state = 82 }		#Banat
				82 = { set_state_flag = HUN_is_being_integrated }
				add_05_conscription_resistance = yes
				add_state_core = 82		#Banat
			}
			if = {
				limit = { owns_state = 83 }		#Southern Partium
				83 = { set_state_flag = HUN_is_being_integrated }
				add_05_conscription_resistance = yes
				add_state_core = 83		#Southern Partium
			}
			if = {
				limit = { owns_state = 84 }		#Transylvania
				84 = { set_state_flag = HUN_is_being_integrated }
				add_05_conscription_resistance = yes
				add_state_core = 84		#Transylvania
			}
			if = {
				limit = { owns_state = 752 }	#Maramaros
				752 = { set_state_flag = HUN_is_being_integrated }
				add_05_conscription_resistance = yes
				add_state_core = 752		#Maramaros
			}
			if = {
				limit = { owns_state = 753 }	#Northern Partium
				753 = { set_state_flag = HUN_is_being_integrated }
				add_05_conscription_resistance = yes
				add_state_core = 753		#Northern Partium
			}
		}
	}
	
	#Don't integrate
	option = {
		name = hungary_flavor_territory.1.b
		custom_effect_tooltip = HUN_transylvania_will_remain_a_territory
		hidden_effect = { set_country_flag = HUN_transylvania_integration_off }
	}
	
	#Decide later
	option = {
		name = hungary_flavor_territory.1.c
		custom_effect_tooltip = HUN_will_get_this_event_again
		hidden_effect = { country_event = { months = 6 id = hungary_flavor_territory.1 } }
	}
}

# Resistance to integration in the Transylvanian region
country_event = {
	id = hungary_flavor_territory.2
	title = hungary_flavor_territory.2.t
	desc = hungary_flavor_territory.2.d
	picture = GFX_report_event_HUN_transylvania
	
	trigger = {
		tag = HUN
		has_country_flag = HUN_transylvania_integration_on
		OR = {
			82 = { has_state_flag = HUN_is_being_integrated }		#Banat
			83 = { has_state_flag = HUN_is_being_integrated }		#Southern Partium
			84 = { has_state_flag = HUN_is_being_integrated }		#Transylvania
			752 = { has_state_flag = HUN_is_being_integrated }		#Maramaros
			753 = { has_state_flag = HUN_is_being_integrated }		#Northern Partium
		}
		NOT = { has_idea = HUN_suppressed_resistance_in_transylvania }
		NOT = { has_idea = HUN_resistance_to_integration_in_transylvania }
	}

	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_01
		}
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_02
		}
		modifier = {
			factor = 0.85
			82 = { has_state_flag = HUN_is_being_integrated }	#Banat
		}
		modifier = {
			factor = 0.85
			83 = { has_state_flag = HUN_is_being_integrated }	#Southern Partium
		}
		modifier = {
			factor = 0.85
			84 = { has_state_flag = HUN_is_being_integrated }	#Transylvania
		}
		modifier = {
			factor = 0.85
			752 = { has_state_flag = HUN_is_being_integrated }	#Maramaros
		}
		modifier = {
			factor = 0.85
			753 = { has_state_flag = HUN_is_being_integrated }	#Northern partium
		}
	}
	
	#We will have to deal with this.
	option = {
		name = hungary_flavor_territory.2.a
		trigger = {
			has_political_power > 24
			has_manpower > 1249
		}
		add_timed_idea = { idea = HUN_suppressed_resistance_in_transylvania days = 365 }
		if = {
			limit = { 82 = { has_state_flag = HUN_is_being_integrated } }	#Banat
			add_manpower = -250
			add_political_power = -5
			82 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Banat
		}
		if = {
			limit = { 83 = { has_state_flag = HUN_is_being_integrated } }	#Southern Partium
			add_manpower = -250
			add_political_power = -5
			83 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Southern Partium
		}
		if = {
			limit = { 84 = { has_state_flag = HUN_is_being_integrated } }	#Transylvania
			add_manpower = -250
			add_political_power = -5
			84 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Transylvania
		}
		if = {
			limit = { 752 = { has_state_flag = HUN_is_being_integrated } }	#Maramaros
			add_manpower = -250
			add_political_power = -5
			752 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Maramaros
		}
		if = {
			limit = { 753 = { has_state_flag = HUN_is_being_integrated } }	#Northern Partium
			add_manpower = -250
			add_political_power = -5
			753 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Northern Partium
		}
	}
	
	#Let this run its course.
	option = {
		name = hungary_flavor_territory.2.b
		add_timed_idea = { idea = HUN_resistance_to_integration_in_transylvania days = 182 }
		custom_effect_tooltip = HUN_integration_in_transylvania_will_slow_down
	}
}

# Rebellion in Banat
country_event = {
	id = hungary_flavor_territory.13
	title = hungary_flavor_territory.13.t
	desc = hungary_flavor_territory.13.d
	picture = GFX_report_event_HUN_transylvania
	
	trigger = {
		tag = HUN
		owns_state = 82		#Banat
		has_country_flag = HUN_transylvania_integration_on
		82 = { has_state_flag = HUN_is_being_integrated }
		NOT = { has_idea = HUN_cracking_down_on_transylvanian_rebels }
	}

	mean_time_to_happen = {
		months = 144
		modifier = {
			factor = 0.75
			has_idea = HUN_suppressed_resistance_in_transylvania
		}
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_01
		}
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_02
		}
		modifier = {
			factor = 1.25
			OR = {
				has_country_flag = HUN_integration_bonus_02
				has_country_flag = HUN_integration_bonus_01
			}
		}
	}
	
	#Let them go.
	option = {
		name = hungary_flavor_territory.13.a
		add_political_power = 25
		custom_effect_tooltip = HUN_integration_of_the_state_will_stop
		remove_05_conscription_resistance = yes
		ROM = { transfer_state = 82 }	#Banat
		remove_state_core = 82			#Banat
		hidden_effect = { 82 = { clr_state_flag = HUN_is_being_integrated } }
	}
	
	#Fight the rebels
	option = {
		name = hungary_flavor_territory.13.b
		custom_effect_tooltip = HUN_integration_of_the_state_will_continue
		random_list = { 
			25 = { add_manpower = -1600 } 
			25 = { add_manpower = -800 } 
			25 = { add_manpower = -400 } 
			25 = { add_manpower = -200 } 
		}
		add_timed_idea = { idea = HUN_cracking_down_on_transylvanian_rebels days = 100 }
		82 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Banat
	}
	
	#Negotiate with the rebels
	option = {
		name = hungary_flavor_territory.13.c
		add_political_power = -25
		custom_effect_tooltip = HUN_integration_of_the_state_will_stop
		remove_05_conscription_resistance = yes
		remove_state_core = 82		#Banat
		hidden_effect = { 
			82 = { 
				clr_state_flag = HUN_is_being_integrated 
				set_state_flag = HUN_integration_abandoned
			} 
		}
	}
}

	id = hungary_flavor_territory.210
	title = hungary_flavor_territory.210.t
	desc = hungary_flavor_territory.210.d
	picture = GFX_report_event_yugoslavian_partition
	
	trigger = {
		tag = HUN
		owns_state = 109 	#Slavonia
		109 = {
			NOT = { is_core_of = ROOT }
			NOT = { has_state_flag = HUN_is_being_integrated }
			NOT = { has_state_flag = HUN_integration_abandoned }
		}
		has_country_flag = HUN_croatia_integration_on
	}

	mean_time_to_happen = {
		days = 0
	}
	
	#Excellent.
	option = {
		name = hungary_flavor_territory.210.a
		custom_effect_tooltip = HUN_integration_of_the_state_will_start
		add_05_conscription_resistance = yes
		add_state_core = 109		#Slavonia
		109 = { set_state_flag = HUN_is_being_integrated }
	}
	
	#The state is fine as it is.
	option = {
		name = hungary_flavor_territory.210.b
		custom_effect_tooltip = HUN_state_will_not_be_integrated
		hidden_effect = { 109 = { set_state_flag = HUN_integration_abandoned } }		#Slavonia
	}
}

# Rebellion in Slavonia
country_event = {
	id = hungary_flavor_territory.213
	title = hungary_flavor_territory.213.t
	desc = hungary_flavor_territory.213.d
	picture = GFX_report_event_yugoslavian_partition
	
	trigger = {
		tag = HUN
		owns_state = 109		#Slavonia
		has_country_flag = HUN_croatia_integration_on
		109 = { has_state_flag = HUN_is_being_integrated }
		NOT = { has_idea = HUN_cracking_down_on_croatian_rebels }
	}

	mean_time_to_happen = {
		months = 144
		modifier = {
			factor = 0.75
			has_idea = HUN_suppressed_resistance_in_croatia
		}
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_01
		}
		modifier = {
			factor = 0.75
			has_country_flag = HUN_suppression_bonus_02
		}
		modifier = {
			factor = 1.25
			OR = {
				has_country_flag = HUN_integration_bonus_02
				has_country_flag = HUN_integration_bonus_01
			}
		}
	}
	
	#Let them go.
	option = {
		name = hungary_flavor_territory.213.a
		add_political_power = 25
		custom_effect_tooltip = HUN_integration_of_the_state_will_stop
		remove_05_conscription_resistance = yes
		CRO = { transfer_state = 109 }	#Slavonia
		remove_state_core = 109			#Slavonia
		hidden_effect = { 109 = { clr_state_flag = HUN_is_being_integrated } }
	}
	
	#Fight the rebels
	option = {
		name = hungary_flavor_territory.213.b
		custom_effect_tooltip = HUN_integration_of_the_state_will_continue
		random_list = { 
			25 = { add_manpower = -1600 } 
			25 = { add_manpower = -800 } 
			25 = { add_manpower = -400 } 
			25 = { add_manpower = -200 } 
		}
		add_timed_idea = { idea = HUN_cracking_down_on_croatian_rebels days = 100 }
		109 = { custom_effect_tooltip = HUN_resistance_will_be_suppressed }	#Slavonia
	}
	
	#Negotiate with the rebels
	option = {
		name = hungary_flavor_territory.213.c
		add_political_power = -25
		custom_effect_tooltip = HUN_integration_of_the_state_will_stop
		remove_05_conscription_resistance = yes
		remove_state_core = 109		#Slavonia
		hidden_effect = { 
			109 = { 
				clr_state_flag = HUN_is_being_integrated 
				set_state_flag = HUN_integration_abandoned
			} 
		}
	}
}
